// ============================================================
//  SECTION 3 вҖ” NADARAYA-WATSON KERNEL REGRESSION (ML CORE)
// ============================================================
//
//  True non-parametric ML estimator.
//  Kernel: Gaussian  K(u) = exp(-uВІ / 2hВІ)
//  Bandwidth: optionally scaled by normalised ATR so the
//  kernel widens in volatile regimes вҶ’ adaptive smoothing.
//
//  Е· = ОЈ K(i) В· close[i]  /  ОЈ K(i)   for i = 0 вҖҰ length-1
//
//  This is evaluated at the current (most recent) bar,
//  treating each historical bar's lag (i) as the distance
//  in the feature space. No future data is used.

// --- adaptive bandwidth computation ---
rawATR     = ta.atr(i_atrLen)            // ATR via shared lib
atrNorm    = rawATR / close              // normalise to price %
atrFactor  = ta.ema(atrNorm, i_atrLen)
// scale: higher vol вҶ’ wider bandwidth (smoother)
adaptScale = i_adaptive ? (1.0 + atrFactor * 200.0) : 1.0
h          = i_bandwidth * adaptScale

// --- kernel regression loop ---
nwSum    = 0.0
nwWeight = 0.0
for i = 0 to i_length - 1
    // Gaussian kernel weight based on lag distance
    kw      = math.exp(-math.pow(i, 2) / (2.0 * math.pow(h, 2)))
    nwSum   += kw * close[i]
    nwWeight += kw

nwRaw = nwSum / nwWeight

// Smooth the output to reduce micro-jitter (still non-repainting)
kernelMA = ta.ema(nwRaw, i_smooth)

// ============================================================
//  SECTION 4 вҖ” RESIDUAL BANDS (Пғ of price vs kernel)
// ============================================================
//
//  Residual = close - kernelMA
//  Пғ  = rolling std-dev of residuals over bandLen
//  Upper band = kernelMA + mult * Пғ
//  Lower band = kernelMA вҲ’ mult * Пғ

residual = close - kernelMA
sigmaRaw = ta.stdev(residual, i_bandLen)
sigma    = ta.ema(sigmaRaw, i_bandSmooth)   // smooth the band width

upperBand = kernelMA + i_bandMult * sigma
lowerBand = kernelMA - i_bandMult * sigma

// ============================================================
//  SECTION 5 вҖ” STATE & SIGNAL (confirmed, non-repainting)
// ============================================================

var int lastState = 0

if barstate.isconfirmed
    if close > upperBand
        lastState := 1
    else if close < lowerBand
        lastState := -1
    // else: hold last state вҖ” no neutral painted

isBull = lastState == 1
isBear = lastState == -1

// crossover/under alerts (use confirmed bar logic)
crossBull = ta.crossover(close,  upperBand)
crossBear = ta.crossunder(close, lowerBand)

// ============================================================
//  STATE COLOR (NO GRADIENT) вҖ” uses template colors

// в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
//  DASHBOARD (styled + subtle vertical gradient)
//  Notes:
//  вҖў create table ONCE (var), then only update cells on last bar
//  вҖў gradient = more visible at header, fades toward bottom
//  вҖў very transparent, nothing loud
// в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ

grad(_base, _row, _rows, _minT, _maxT) =>
    // _row: 0.._rows-1

    // Row 2
    table.cell(tbl, 0, 2, "Kernel MA",
         text_color=C_TEXT, text_size=size.normal, bgcolor=bg2,
         text_halign=text.align_left, text_font_family=font.family_monospace)
    table.cell(tbl, 1, 2, str.tostring(kernelMA, "#.##"),
         text_color=solidCol, text_size=size.normal, bgcolor=bg2,
         text_halign=text.align_right, text_font_family=font.family_monospace)

    // Row 3
    table.cell(tbl, 0, 3, "Upper Band",
         text_color=C_TEXT, text_size=size.normal, bgcolor=bg3,

    table.cell(tbl, 1, 3, str.tostring(upperBand, "#.##"),
         text_color=C_BULL, text_size=size.normal, bgcolor=bg3,
         text_halign=text.align_right, text_font_family=font.family_monospace)

    // Row 4
    table.cell(tbl, 0, 4, "Lower Band",
         text_color=C_TEXT, text_size=size.normal, bgcolor=bg4,
         text_halign=text.align_left, text_font_family=font.family_monospace)
    table.cell(tbl, 1, 4, str.tostring(lowerBand, "#.##"),
         text_color=C_BEAR, text_size=size.normal, bgcolor=bg4,
         text_halign=text.align_right, text_font_family=font.family_monospace)

    // Row 5
    table.cell(tbl, 0, 5, "Band Width Пғ",
         text_color=C_TEXT, text_size=size.normal, bgcolor=bg5,
         text_halign=text.align_left, text_font_family=font.family_monospace)
    table.cell(tbl, 1, 5, str.tostring(sigma, "#.##"),
         text_color=C_TEXT, text_size=size.normal, bgcolor=bg5,
         text_halign=text.align_right, text_font_family=font.family_monospace)

    // Row 6
    bwLabel = i_adaptive ? str.tostring(h, "#.##") + "  (adaptive)" : str.tostring(h, "#.##")
    table.cell(tbl, 0, 6, "Bandwidth h",
         text_color=C_TEXT, text_size=size.normal, bgcolor=bgBot,
         text_halign=text.align_left, text_font_family=font.family_monospace)
    table.cell(tbl, 1, 6, bwLabel,
         text_color=C_TEXT, text_size=size.normal, bgcolor=bgBot,
         text_halign=text.align_right, text_font_family=font.family_monospace)
